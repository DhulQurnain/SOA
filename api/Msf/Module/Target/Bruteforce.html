def default_direction
  dd = self['DefaultDirection']

  if (dd and dd.to_s.match(/(-1|backward)/i))
    return -1
  end

  return 1
end

def start_addresses
  if (self['Start'] and self['Start'].kind_of?(Hash) == false)
    return {'Address' => self['Start'] }
  else
    return self['Start']
  end
end
