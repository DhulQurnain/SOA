def get_build_prop
  sys_data   = {}
  build_prop = cmd_exec('cat /system/build.prop')

  return sys_data if build_prop.blank?

  build_prop.scan(/(.+)=(.+)/i).collect {|e| Hash[*e]}.each do |setting|
    sys_data.merge!(setting)
  end

  return sys_data
end
