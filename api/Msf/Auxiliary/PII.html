def luhnCheck(ccNumber)
  ccNumber = ccNumber.gsub(/\D/, '')
  cardLength = ccNumber.length
  parity = cardLength % 2
  sum = 0
  for i in 0...cardLength
    digit = ccNumber[i] - 48
    if i % 2 == parity
      digit = digit * 2
    end
    if digit > 9
      digit = digit - 9
    end
    sum = sum + digit
  end
  return (sum % 10)
end
