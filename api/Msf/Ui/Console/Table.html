def initialize(style, opts = {})
  self.style = style

  if (self.style == Style::Default)
    opts['Indent']  = 3
    if (!opts['Prefix'])
      opts['Prefix']  = "\n"
    end
    if (!opts['Postfix'])
      opts['Postfix'] = "\n"
    end

    super(opts)
  end
end
