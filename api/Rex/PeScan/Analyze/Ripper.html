def scan(param)
  dest = param['dir']

  if (param['file'])
    dest = File.join(dest, File.basename(param['file']))
  end

  ::FileUtils.mkdir_p(dest)

  pe.resources.keys.sort.each do |rkey|
    res  = pe.resources[rkey]
    path = File.join(dest, rkey.split('/')[1] + '_' + res.file)

    fd = File.new(path, 'wb')
    fd.write(res.data)
    fd.close
  end
end
