def create(arg, opt ={})
  t = new
  t.lm_response = arg[:lm_response]
  t.ntlm_response = arg[:ntlm_response]
  t.domain = arg[:domain]
  t.user = arg[:user]
  t.workstation = arg[:workstation]

  if arg[:session_key]
    t.enable(:session_key)
    t.session_key = arg[session_key]
  end
  if arg[:flag]
    t.enable(:session_key)
    t.enable(:flag)
    t.flag = arg[:flag]
  end
  t
end
